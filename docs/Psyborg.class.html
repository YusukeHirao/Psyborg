  getBackgroundColor = ($elem) ->
    bgColorValue = $elem.css 'background-color'
    if bgColorValue.indexOf('r') is 0
      rgba = bgColorValue
        .replace(/\s+/ig, '') # 空白の除去
        .replace /^rgba?\(([0-9,.%]+)\)/, '$1' # rgba() の除去
      rgba = rgba.split(',')
      for color, i in rgba
        rgba[i] = int color
      if rgba.length is 3
        rgba.push 1
      return rgba
    else if bgColorValue.indexOf('#') is 0
      hex = bgColorValue.substring 1 # '#'の除去
      if hex.length is 3
        hex = hex.replace /^([0-9a-f])([0-9a-f])([0-9a-f])$/i, '$1$1$2$2$3$3'
      r = hex.substring 0, 2
      g = hex.substring 2, 4
      b = hex.substring 4, 6
      r = parseInt r, 16
      g = parseInt g, 16
      b = parseInt b, 16
      return [r, g, b, 1]
    else
      return [0, 0, 0, 0]
